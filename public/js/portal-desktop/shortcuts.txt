	dameShortCuts: function(objson){
		if (objson && objson.length){
			for(var i=0; i<objson.length;i++){
				if (objson[i].menu){
					dameShortCuts(objson[i].menu);
				}
				if (objson[i].referencia){
						arrayShortCuts.push({
						id: objson[i].id,
						text: objson[i].text,
						ref: objson[i].referencia
					});
				}
			} // for

	}
